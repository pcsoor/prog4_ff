199  public Brush GetBrush(string fname, bool isTiled)
200  {
201  if (!this.myBrushes.ContainsKey(fname))
202  {
203  BitmapImage bmp = new BitmapImage();
204  bmp.BeginInit();
205  bmp.StreamSource = Assembly.GetExecutingAssembly().GetManifestResourceStream(fname);
206  bmp.EndInit();
207  ImageBrush ib = new ImageBrush(bmp);
208  if (isTiled)
209  {
210  ib.TileMode = TileMode.Tile;
211  ib.Viewport = new Rect(0, 0, this.model.TileSize, this.model.TileSize);
212  ib.ViewportUnits = BrushMappingMode.Absolute;
213  }
214 
215  this.myBrushes[fname] = ib;
216  }
217 
218  return this.myBrushes[fname];
219  }

225  public void BuildDisplay(DrawingContext ctx)
226  {
227  if (ctx != null)
228  {
229  for (int x = 0; x < GameModel.Map.GetLength(0); x++)
230  {
231  for (int y = 0; y < GameModel.Map.GetLength(1); y++)
232  {
233  if (GameModel.Map[x, y] != null)
234  {
235  Geometry box = new RectangleGeometry(new Rect(x * this.model.TileSize, y * this.model.TileSize, this.model.TileSize, this.model.TileSize));
236  if (GameModel.Map[x, y].Type == Enumerators.Types.Block)
237  {
238  if (GameModel.Map[x, y].Color == 1)
239  {
240  ctx.DrawGeometry(this.BlueBlockBrush, null, box);
241  }
242  else if (GameModel.Map[x, y].Color == 2)
243  {
244  ctx.DrawGeometry(this.GreenBlockBrush, null, box);
245  }
246  else if (GameModel.Map[x, y].Color == 3)
247  {
248  ctx.DrawGeometry(this.GoldBlockBrush, null, box);
249  }
250  else if (GameModel.Map[x, y].Color == 4)
251  {
252  ctx.DrawGeometry(this.GreyBlockBrush, null, box);
253  }
254  else if (GameModel.Map[x, y].Color == 5)
255  {
256  ctx.DrawGeometry(this.BrownBlockBrush, null, box);
257  }
258  }
259 
260  if (GameModel.Map[x, y].Type == Enumerators.Types.Metal)
261  {
262  if (GameModel.Map[x, y].Color == 1)
263  {
264  ctx.DrawGeometry(this.BlueMetalBrush, null, box);
265  }
266  else if (GameModel.Map[x, y].Color == 2)
267  {
268  ctx.DrawGeometry(this.GreenMetalBrush, null, box);
269  }
270  else if (GameModel.Map[x, y].Color == 3)
271  {
272  ctx.DrawGeometry(this.GoldMetalBrush, null, box);
273  }
274  else if (GameModel.Map[x, y].Color == 4)
275  {
276  ctx.DrawGeometry(this.GreyMetalBrush, null, box);
277  }
278  else if (GameModel.Map[x, y].Color == 5)
279  {
280  ctx.DrawGeometry(this.BrownMetalBrush, null, box);
281  }
282  }
283 
284  if (GameModel.Map[x, y].Type == Enumerators.Types.PowerUp)
285  {
286  ctx.DrawGeometry(this.PowerUpBrush, null, box);
287  }
288 
289  if (GameModel.Map[x, y].Type == Enumerators.Types.Wall)
290  {
291  ctx.DrawGeometry(this.WallBrush, null, box);
292  }
293 
294  if (GameModel.Map[x, y].Type == Enumerators.Types.Player)
295  {
296  ctx.DrawGeometry(this.PlayerBrush, null, box);
297  }
298 
299  if (GameModel.Map[x, y].Type == Enumerators.Types.Bullet)
300  {
301  ctx.DrawGeometry(this.BulletBrush, null, box);
302  }
303  }
304  }
305  }
306 
307  this.DrawLifeText(ctx);
308  this.DrawBlockStormActive(ctx);
309  this.DrawMetalActive(ctx);
310  this.DrawDoubleJumpActive(ctx);
311  this.DrawDoublePushActive(ctx);
312  this.Highscore(ctx);
313  }
